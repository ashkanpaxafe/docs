---
title: TemperatureRange
---

<ResponseField name="pte" type="string or null">
  Predicted Temperature Excursion status
  <Expandable title="Examples">
    - `normal temperature`
    - `alarm: temp below low_limit`
    - `alarm: temp above up_limit`
    - `warning: temp approaching low_limit`
    - `warning: temp approaching up_limit`
  </Expandable>
</ResponseField>

<ResponseField name="excursion" type="boolean">
  Whether a temperature excursion has occurred
  <Expandable title="Example">
    `false`
  </Expandable>
</ResponseField>

<ResponseField name="time_out_of_range" type="integer">
  Time spent out of the acceptable temperature range in minutes
  <Expandable title="Example">
    `540`
  </Expandable>
</ResponseField>

<ResponseField name="remaining_time_out_of_range" type="integer">
  Remaining time allowed out of the acceptable temperature range in minutes
  <Expandable title="Example">
    `60`
  </Expandable>
</ResponseField>

<ResponseField name="remaining_shelf_life" type="string">
  Remaining shelf life of the product
  <Expandable title="Example">
    `10%`
  </Expandable>
</ResponseField>

<ResponseField name="num_excursions" type="integer">
  Number of temperature excursions that have occurred
</ResponseField>

<ResponseField name="low_limit" type="integer">
  Lower limit of Product temperature range
</ResponseField>

<ResponseField name="up_limit" type="integer">
  Upper limit of Product temperature range
  <Expandable title="Example">
    `8`
  </Expandable>
</ResponseField>

<ResponseExample>
  ```json Example
  {
  "pte": "normal temperature",
  "excursion": false,
  "time_out_of_range": 540,
  "remaining_time_out_of_range": 60,
  "remaining_shelf_life": "10%",
  "num_excursions": 0,
  "low_limit": 0,
  "up_limit": 8
}
  ```
</ResponseExample>
